25 static uint8_t genCRC(uint8_t *input, int lenInput)
26 {
27 uint8_t generator = 0x07;
28 uint8_t crc = 0;
29
30 for (int x = 0; x < lenInput; x++)
31 {
32 crc ^= input[x]; /* XOR-in the next input byte */
33
34 for (int i = 0; i < 8; i++)
35 {
36 if ((crc & 0x80) != 0)

38 crc = (uint8_t)((crc << 1) ^ generator);
39 }

42 crc <<= 1;
43 }
44 }
45 }
